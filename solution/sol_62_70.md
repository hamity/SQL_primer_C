```sql
select 経験イベント.ルート番号, 経験イベント.イベント番号, イベント.イベント名称, 経験イベント.クリア結果
from 経験イベント
left join イベント
on 経験イベント.イベント番号 = イベント.イベント番号
where 経験イベント.クリア結果 is not null
order by 経験イベント.ルート番号;       -- 62

select イベント.イベント番号, イベント.イベント名称, 経験イベント.クリア区分
from 経験イベント
left join イベント
on 経験イベント.イベント番号 = イベント.イベント番号
where イベント.タイプ in ('3'); -- 63

select イベント.イベント番号, イベント.イベント名称, coalesce(経験イベント.クリア区分, '未クリア')
from イベント
left join 経験イベント
on 経験イベント.イベント番号 = イベント.イベント番号
where イベント.タイプ in ('3');  -- 64
```
